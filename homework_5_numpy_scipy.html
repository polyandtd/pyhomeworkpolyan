      def create_array(n):
          result =[]
          while n >= 0:
              result.append(n)
              n-=1
          a = np.array(result)
          return a

      def diag_matrix(n):
          a = np.diag(np.arange(n))
          cnt = 0
          while n >= 0:
              n-=1
              cnt+= n
          cnt+=1 #в таком подсчёте не хватает единицы
          b = a[::-1, ::-1]
          print(cnt)
          return b
